function convertTextToHTML(text) {
    const lines = text.split('\n');
    let result = '';
    let inList = false;

    lines.forEach(line => {
        line = line.trim();
        if (line.startsWith('#')) {
            if (inList) {
                result += '</ul>\n';
                inList = false;
            }
            else if (!inList) {
                result += '<ul>\n';
                inList = true;
            }
        } else if (line.startsWith('-')) {
            result += `<li>${line.substring(1).trim()}</li>\n`;
        }
    });

    if (inList) {
        result += '</ul>';
    }

    return result;
}
